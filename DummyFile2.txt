<ItemGroup>
  <None Include="Scripts\PreDeploy_BreakingChanges.sql">
    <CopyToOutputDirectory>PreserveNewest</CopyToOutputDirectory>
  </None>
</ItemGroup>

<Target Name="CopyManifestFilesToOutput" BeforeTargets="PrepareForRun">

  <ReadLinesFromFile File="Scripts\PreDeploy_BreakingChanges.sql">
    <Output TaskParameter="Lines" ItemName="_RawManifestLines" />
  </ReadLinesFromFile>

  <ItemGroup>
    <_ManifestPaths Include="@(_RawManifestLines)"
                    Condition="$([System.Text.RegularExpressions.Regex]::IsMatch('%(Identity)', '^\s*:r\s+'))">
      
      <CleanPath>$([System.Text.RegularExpressions.Regex]::Replace('%(Identity)', '^\s*:r\s+[&quot;&apos;]?(\.\\)?', ''))</CleanPath>
      
      <CleanPath>%(CleanPath.Replace('%0D', '').Replace('%0A', ''))</CleanPath>
      
      <CleanPath>%(CleanPath.Trim().Trim('&quot;').Trim("'"))</CleanPath>
    </_ManifestPaths>
  </ItemGroup>

  <Message Importance="High" Text="--- DEBUG MANIFEST START ---" />
  <Message Importance="High" Text="Found Script: [%(_ManifestPaths.CleanPath)]" />
  <Message Importance="High" Text="Full Source Path: [$(MSBuildProjectDirectory)\Scripts\%(_ManifestPaths.CleanPath)]" />
  <Message Importance="High" Text="--- DEBUG MANIFEST END ---" />

  <Error Condition="'@(_ManifestPaths)' != '' AND '@(_ManifestPaths->'%(CleanPath)')' != '@(_ManifestPaths->'%(CleanPath)'->Distinct())'"
         Text="Duplicate entries found in PreDeploy_BreakingChanges.sql - please clean up the manifest." />

  <Error Condition="'@(_ManifestPaths)' != '' AND !Exists('$(MSBuildProjectDirectory)\Scripts\%(_ManifestPaths.CleanPath)')"
         Text="Build Error: Breaking change script not found at: [$(MSBuildProjectDirectory)\Scripts\%(_ManifestPaths.CleanPath)]" />

  <Copy Condition="'@(_ManifestPaths)' != ''"
        SourceFiles="@(_ManifestPaths->'$(MSBuildProjectDirectory)\Scripts\%(CleanPath)')"
        DestinationFiles="@(_ManifestPaths->'$(TargetDir)Scripts\%(CleanPath)')"
        SkipUnchangedFiles="true" />

</Target>
