<ItemGroup>
  <None Include="Scripts\PreDeploy_BreakingChanges.sql">
    <CopyToOutputDirectory>PreserveNewest</CopyToOutputDirectory>
  </None>
</ItemGroup>

<Target Name="CopyManifestFilesToOutput" BeforeTargets="PrepareForRun">

  <ReadLinesFromFile File="Scripts\PreDeploy_BreakingChanges.sql">
    <Output TaskParameter="Lines" ItemName="_RawManifestLines" />
  </ReadLinesFromFile>

  <ItemGroup>
    <_ManifestPaths Include="@(_RawManifestLines)"
                    Condition="$([System.Text.RegularExpressions.Regex]::IsMatch('%(Identity)', '^\s*:r\s+'))">
      <RelativePath>$([System.Text.RegularExpressions.Regex]::Replace('%(Identity)', '^\s*:r\s+[&quot;&apos;]?(\.\\)?', '').Trim().Trim('&quot;').Trim(&quot;'&quot;))</RelativePath>
    </_ManifestPaths>
  </ItemGroup>

  <Error Condition="'@(_ManifestPaths)' != '' AND '@(_ManifestPaths->'%(RelativePath)')' != '@(_ManifestPaths->'%(RelativePath)'->Distinct())'"
         Text="Duplicate entries found in PreDeploy_BreakingChanges.sql" />

  <Error Condition="'@(_ManifestPaths)' != '' AND !Exists('$(MSBuildProjectDirectory)\%(_ManifestPaths.RelativePath)')"
         Text="Build Error: Breaking change script not found: $(MSBuildProjectDirectory)\%(_ManifestPaths.RelativePath)" />

  <Copy Condition="'@(_ManifestPaths)' != ''"
        SourceFiles="@(_ManifestPaths->'$(MSBuildProjectDirectory)\%(RelativePath)')"
        DestinationFiles="@(_ManifestPaths->'$(TargetDir)PreDeploy_BreakingChanges\%(RelativePath)')"
        SkipUnchangedFiles="true" />

</Target>
