<ItemGroup>
  <None Include="Scripts\PreDeploy_BreakingChanges.sql">
    <CopyToOutputDirectory>PreserveNewest</CopyToOutputDirectory>
  </None>
</ItemGroup>

<Target Name="CopyManifestFilesToOutput" BeforeTargets="PrepareForRun">

  <ReadLinesFromFile File="Scripts\PreDeploy_BreakingChanges.sql">
    <Output TaskParameter="Lines" ItemName="_RawManifestLines" />
  </ReadLinesFromFile>

  <ItemGroup>
    <_ManifestPaths Include="@(_RawManifestLines)"
                    Condition="$([System.Text.RegularExpressions.Regex]::IsMatch('%(Identity)', '^\s*:r\s+'))">
      <RawPath>$([System.Text.RegularExpressions.Regex]::Replace('%(Identity)', '^\s*:r\s+[&quot;&apos;]?(\.\\)?', ''))</RawPath>
    </_ManifestPaths>

    <_CleanManifestPaths Include="@(_ManifestPaths)">
      <CleanPath>$([System.String]::new('%(RawPath)').Replace('%0D', '').Replace('%0A', '').Trim().Trim('&quot;').Trim("'"))</CleanPath>
    </_CleanManifestPaths>
  </ItemGroup>

  <Message Importance="High" Text="--- DEBUG MANIFEST START ---" />
  <Message Importance="High" Text="Cleaned Filename: [%(_CleanManifestPaths.CleanPath)]" />
  <Message Importance="High" Text="Full Source Path: [$(MSBuildProjectDirectory)\Scripts\%(_CleanManifestPaths.CleanPath)]" />
  <Message Importance="High" Text="--- DEBUG MANIFEST END ---" />

  <Error Condition="'@(_CleanManifestPaths)' != '' AND '@(_CleanManifestPaths->'%(CleanPath)')' != '@(_CleanManifestPaths->'%(CleanPath)'->Distinct())'"
         Text="Duplicate entries found in PreDeploy_BreakingChanges.sql." />

  <Error Condition="'@(_CleanManifestPaths)' != '' AND !Exists('$(MSBuildProjectDirectory)\Scripts\%(_CleanManifestPaths.CleanPath)')"
         Text="Build Error: Breaking change script not found at: [$(MSBuildProjectDirectory)\Scripts\%(_CleanManifestPaths.CleanPath)]" />

  <Copy Condition="'@(_CleanManifestPaths)' != ''"
        SourceFiles="@(_CleanManifestPaths->'$(MSBuildProjectDirectory)\Scripts\%(CleanPath)')"
        DestinationFiles="@(_CleanManifestPaths->'$(TargetDir)Scripts\%(CleanPath)')"
        SkipUnchangedFiles="true" />

</Target>
