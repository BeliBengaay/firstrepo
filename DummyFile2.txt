<ItemGroup>
  <None Include="Scripts\PreDeploy_BreakingChanges.sql">
    <CopyToOutputDirectory>PreserveNewest</CopyToOutputDirectory>
  </None>
</ItemGroup>

<Target Name="CopyManifestFilesToOutput" BeforeTargets="PrepareForRun">

  <ReadLinesFromFile File="Scripts\PreDeploy_BreakingChanges.sql">
    <Output TaskParameter="Lines" ItemName="_RawManifestLines" />
  </ReadLinesFromFile>

  <ItemGroup>
    <_ManifestPaths Include="$([System.Text.RegularExpressions.Regex]::Replace('%(_RawManifestLines.Identity)', '^\s*:r\s+[&quot;&apos;]?(\.\\)?', '').Trim().Trim('&quot;').Trim(&quot;'&quot;))"
                    Condition="$([System.Text.RegularExpressions.Regex]::IsMatch('%(_RawManifestLines.Identity)', '^\s*:r\s+'))" />
  </ItemGroup>

  <Error Condition="'@(_ManifestPaths)' != '' AND '@(_ManifestPaths)' != '@(_ManifestPaths->Distinct())'"
         Text="Duplicate entries found in PreDeploy_BreakingChanges.sql - please clean up the manifest." />

  <ItemGroup Condition="'@(_ManifestPaths)' != ''">
    <_ResolvedPaths Include="@(_ManifestPaths)">
      <ResolvedFullPath>$([System.IO.Path]::GetFullPath('$(MSBuildProjectDirectory)\%(Identity)'))</ResolvedFullPath>
      <RelativePath>%(Identity)</RelativePath>
    </_ResolvedPaths>
  </ItemGroup>

  <Error Condition="'@(_ResolvedPaths)' != '' AND !Exists('%(_ResolvedPaths.ResolvedFullPath)')"
         Text="Build Error: Breaking change script not found: %(_ResolvedPaths.ResolvedFullPath)" />

  <Copy Condition="'@(_ResolvedPaths)' != ''"
        SourceFiles="@(_ResolvedPaths->'%(ResolvedFullPath)')"
        DestinationFiles="@(_ResolvedPaths->'$(TargetDir)PreDeploy_BreakingChanges\%(RelativePath)')"
        SkipUnchangedFiles="true" />

</Target>
